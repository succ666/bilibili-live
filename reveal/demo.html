	// 组件内没有内容
	<empty-component />

	// 组件标签之间有内容
	<container-component>
	    This is content.
	</container-component>
</template>

<script>
import EmptyComponent from '../path';
import ContainerComponent from '../../path';

export default {
	name: 'MyComponent',
	components: {
		EmptyComponent,
		ContainerComponent
	}
}
</script>
